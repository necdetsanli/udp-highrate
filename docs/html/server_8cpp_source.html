162
163* @brief Main worker loop: batch receive, optional echo, rate logging, metrics updates.
164
165*
166
167* @details
168
169* Receive & counters:
170
171* - Allocates `cfg_.batch` receive buffers (2048 bytes each) and calls
172
173* `ISocket::recv_batch(bufs)` repeatedly.
174
175* - For each received message:
176
177* - Optionally inspects a `PacketHeader` (if the payload contains one) for
178
179* basic integrity (magic). This example doesnâ€™t extract peer addresses.
180
181* - Updates hot-path counters: `inc_recv(1)` and `add_rx_bytes(size)`.
182
183*
184

186
187* - When `cfg_.echo` is true, reuses the received buffers as a batch to send
188
189* back via `send_batch`. On success, updates `sent` and `tx_bytes` counters.
190

192

194
195* - Every ~1s, computes delta of `stats_.recv()` since the last tick and
196

204
205* - If `recv_batch` returns < 0, the loop continues (best-effort behavior).
206
207* - Non-blocking sockets may frequently return 0 (no messages available).
208

211void UdpServer::run_loop() {
212
213 std::vector<std::vector<uint8_t>> bufs(cfg_.batch, std::vector<uint8_t>(2048));
214
215 uint64_t last_recv_total = 0;
216
217 auto last_ts = std::chrono::steady_clock::now();
218
219 while (running_) {
220
221 ssize_t r = sock_->recv_batch(bufs);
222
223 if (r < 0) continue;
224
225 if (r > 0) {
226
227 for (ssize_t i=0;i<r;i++) {
228
229 // Track client from a fake header: in real world we would read src addr from recvmmsg
230
231 // Here we approximate by requiring clients to include magic header at start
232
233 if (bufs[i].size() >= sizeof(PacketHeader)) {
234
235 PacketHeader* hdr = reinterpret_cast<PacketHeader*>(bufs[i].data());
236
237 if (hdr->magic == kMagic) {
238
239 // Cannot access peer addr without msghdr name here (already set in socket), so skip addr track
240
241 }
242
243 }
244
245 stats_.inc_recv(1);
246
247 stats_.add_rx_bytes(bufs[i].size());
248
249 }
250
251 if (cfg_.echo) {
252
253 // Echo back
254
255 std::vector<std::vector<uint8_t>> out;
256
257 out.reserve(r);
258
259 for (ssize_t i=0;i<r;i++) out.push_back(bufs[i]);
260
261 ssize_t s = sock_->send_batch(out, nullptr);
262
263 if (s > 0) {
264
265 stats_.inc_sent(s);
266
267 size_t total_bytes = 0; for (auto& b: out) total_bytes += b.size();
268
269 stats_.add_tx_bytes(total_bytes);
270
271 }
272
273 }
274
275 }
276
277 auto now = std::chrono::steady_clock::now();
278
279 if (now - last_ts >= std::chrono::seconds(1)) {
280
281 uint64_t recv_total = stats_.recv();
282
283 uint64_t delta = recv_total - last_recv_total;
284
285 last_rate_pps_ = static_cast<double>(delta);
286
287 if (cfg_.verbose) {
288
289 std::cout << "[server] " << stats_.to_string()
290<< " rate=" << human_rate(last_rate_pps_) << "\n";
291
292 }
293
294 last_recv_total = recv_total;
295
296 last_ts = now;
297
298 }
299
300 }
