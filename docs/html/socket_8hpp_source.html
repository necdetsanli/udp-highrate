10/**
11* @file
12* @brief Socket abstraction for high-rate UDP I/O (batch send/receive) plus a test double.
13*
14* This header defines:
15* - @ref udp::ISocket : the strategy/port interface that core logic depends on,
16* - @ref udp::UdpSocket : a concrete Linux/POSIX implementation optimized for batch I/O,
17* - @ref udp::MockSocket : a lightweight in-memory test double.
18*
19* The goal is to decouple business logic (client/server) from the OS-specific
20* socket details, making the system testable (via dependency injection) and
21* evolvable (e.g., future io_uring/DPDK adapters) without touching the core.
22*
23* @note Thread-safety: unless otherwise stated, instances are not designed for
24* concurrent calls into the same object from multiple threads. Prefer
25* single-threaded ownership per @ref ISocket instance, or synchronize externally.

30/**
31* @brief Abstract socket interface (strategy/port).
32*
33* Core code (client/server) talks to sockets solely through this interface.
34* Concrete implementations (e.g., @ref UdpSocket, @ref MockSocket) realize it.
35*
36* @par Batch semantics
37* - @ref recv_batch and @ref send_batch operate on vectors of contiguous buffers.
38* - Implementations typically attempt to process up to @c bufs.size() messages in
39* one call (e.g., via @c recvmmsg/@c sendmmsg on Linux) and return the number of
40* messages successfully processed (not bytes).
41*/
42class ISocket {
43public:
44 virtual ~ISocket() = default;
45
46 /**
47 * @brief File descriptor for low-level polling/integration.
48 * @return Underlying file descriptor, or -1 if not applicable (e.g., @ref MockSocket).
49 */
50 virtual int fd() const = 0;
51
52 /**
53 * @brief Bind the socket to a local UDP port.
54 *
55 * @param port Local UDP port in host byte order.
56 * @param reuseport If true, attempt to enable @c SO_REUSEPORT to allow multi-worker setups.
57 *
58 * @throws std::system_error (typical in implementations) on failure.
59 */
60 virtual void bind(uint16_t port, bool reuseport) = 0;
61
62 /**
63 * @brief Optionally connect the socket to a fixed remote endpoint.
64 *
65 * After a successful connect, datagrams can be sent/received without specifying
66 * a destination address on each call. Some implementations may still support
67 * per-call destination via the @p addr parameter of @ref send_batch.
68 *
69 * @param ip Remote IPv4 address as dotted string (e.g., "127.0.0.1").
70 * @param port Remote UDP port in host byte order.
71 *
72 * @throws std::system_error on failure.
73 */
74 virtual void connect(const std::string& ip, uint16_t port) = 0;
75
76 /**
77 * @brief Receive up to @c bufs.size() datagrams in a single call.
78 *
79 * Each element in @p bufs must be a pre-sized buffer (capacity = max datagram size).
80 * Implementations write up to the buffer size for each received message.
81 *
82 * @param bufs Vector of destination buffers to fill (one datagram per buffer).
83 * @return Number of datagrams received (>= 0), or -1 on error (with errno set).
84 *
85 * @note Return value counts messages, not bytes. If a datagram exceeds a bufferâ€™s
86 * size, implementations may truncate it (consistent with @c recvfrom semantics).
87 */
88 virtual ssize_t recv_batch(std::vector<std::vector<uint8_t>>& bufs) = 0;
89
90 /**
91 * @brief Send up to @c bufs.size() datagrams in a single call.
92 *
93 * @param bufs Vector of source buffers (one datagram per element).
94 * @param addr Optional destination. If non-null, sent to this peer for all
95 * buffers in the batch. If null and the socket is connected,
96 * the connected peer is used. If null and not connected, behavior
97 * is implementation-defined (likely an error).
98 * @return Number of datagrams sent (>= 0), or -1 on error (with errno set).
99 *
100 * @note Return value counts messages, not bytes. Implementations try to minimize
101 * syscalls (e.g., @c sendmmsg) but may fall back to per-message sends.
102 */
103 virtual ssize_t send_batch(const std::vector<std::vector<uint8_t>>& bufs,
104 const sockaddr_in* addr = nullptr) = 0;
105
106 /**
107 * @brief Hint the desired receive buffer size (bytes).
108 * @param bytes Requested size in bytes for @c SO_RCVBUF.
109 *
110 * @note Implementations may clamp or ignore values depending on OS limits.
111 */
112 virtual void set_rcvbuf(int bytes);
113
114 /**
115 * @brief Hint the desired send buffer size (bytes).
116 * @param bytes Requested size in bytes for @c SO_SNDBUF.
117 *
118 * @note Implementations may clamp or ignore values depending on OS limits.
119 */
120 virtual void set_sndbuf(int bytes);
121};
