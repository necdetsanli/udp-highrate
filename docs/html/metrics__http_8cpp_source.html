181/**
182
183* @brief Accept loop: bind/listen on 127.0.0.1:port_, serve a single 200 OK per connection.
184
185*
186
187* @details
188
189* Socket setup:
190
191* - Creates a TCP socket, enables `SO_REUSEADDR` (and `SO_REUSEPORT` if available).
192
193* - Binds to loopback (`127.0.0.1`) only; listens with a small backlog.
194
195*
196
197* Loop semantics:
198
199* - While @ref running_ is true: `accept()` one client; on failure sleeps briefly
200
201* (50 ms) and retries.
202
203* - For each accepted client:
204
205* 1. Build the metrics body via @ref render().
206
207* 2. Write a minimal HTTP/1.1 200 OK response with `Content-Type: text/plain`
208
209* and `Content-Length`.
210
211* 3. Close the client socket.
212
213*
214
215* Shutdown:
216
217* - When @ref running_ becomes false, closes the listening socket and exits.
218
219*/
220
221void MetricsHttpServer::run() {
222
223 int s = ::socket(AF_INET, SOCK_STREAM, 0);
224
225 int one = 1;
226
227 setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
228
229#ifdef SO_REUSEPORT
230
231 setsockopt(s, SOL_SOCKET, SO_REUSEPORT, &one, sizeof(one));
232
233#endif
234
235 sockaddr_in addr{};
236
237 addr.sin_family = AF_INET;
238
239 addr.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
240
241 addr.sin_port = htons(port_);
242
243 bind(s, (sockaddr*)&addr, sizeof(addr));
244
245 listen(s, 8);
246
247 while (running_) {
248
249 sockaddr_in peer{};
250
251 socklen_t plen=sizeof(peer);
252
253 int c = accept(s, (sockaddr*)&peer, &plen);
254
255 if (c < 0) { std::this_thread::sleep_for(std::chrono::milliseconds(50)); continue; }
256
257 std::string body = render();
258
259 std::ostringstream resp;
260
261 resp << "HTTP/1.1 200 OK\r\nContent-Type: text/plain\r\nContent-Length: " << body.size()
262<< "\r\nConnection: close\r\n\r\n" << body;
263
264 auto sstr = resp.str();
265
266 (void)send(c, sstr.data(), sstr.size(), 0);
267
268 close(c);
269
270 }
271
272 close(s);
273
274}
