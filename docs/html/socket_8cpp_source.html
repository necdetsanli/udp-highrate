73/**
74
75* @brief Create a non-blocking UDP socket (IPv4).
76
77*
78
79* @details
80
81* - Uses `AF_INET`/`SOCK_DGRAM` with protocol 0.
82
83* - Sets `O_NONBLOCK` so that hot paths can poll and treat `EAGAIN`/`EWOULDBLOCK`
84
85* as "no messages" (return 0) rather than blocking the thread.
86
87*
88
89* @throws std::runtime_error if `socket()` fails.
90
91*/
92
93static int make_socket() {
94
95 int s = ::socket(AF_INET, SOCK_DGRAM, 0);
96
97 if (s < 0) throw std::runtime_error("socket() failed");
98
99 int flags = fcntl(s, F_GETFL, 0);
100
101 fcntl(s, F_SETFL, flags | O_NONBLOCK);
102
103 return s;
104
105}

225/**
226
227* \copydoc udp::ISocket::recv_batch
228
229*
230
231* @details Linux fast-path:
232
233* - Prepares `mmsghdr`/`iovec` arrays for up to `bufs.size()` messages.
234
235* - Calls `recvmmsg()` once; on `EAGAIN`/`EWOULDBLOCK` returns 0 (no messages).
236
237* - On other errors returns -1; otherwise returns the number of messages received.
238
239*
240
241* Fallback:
242
243* - Performs a single `recvfrom` into `bufs[0]` and returns 1 on success,
244
245* 0 on `EAGAIN`/`EWOULDBLOCK`, or -1 on error.
246
247*
248
249* @note Return value is a **message count**, not bytes.
250
251*/
252
253ssize_t UdpSocket::recv_batch(std::vector<std::vector<uint8_t>>& bufs) {
254
255#if defined(__linux__)
256
257 // Use recvmmsg if available
258
259 const size_t n = bufs.size();
260
261 std::vector<iovec> iov(n);
262
263 std::vector<mmsghdr> msgs(n);
264
265 std::vector<sockaddr_in> addrs(n);
266
267 std::vector<char> ctrl(64 * n); // ancillary control (unused), keeps API parity
268
269 for (size_t i=0;i<n;i++) {
270
271 iov[i].iov_base = bufs[i].data();
272
273 iov[i].iov_len = bufs[i].size();
274
275 memset(&msgs[i], 0, sizeof(mmsghdr));
276
277 msgs[i].msg_hdr.msg_iov = &iov[i];
278
279 msgs[i].msg_hdr.msg_iovlen = 1;
280
281 msgs[i].msg_hdr.msg_name = &addrs[i];
282
283 msgs[i].msg_hdr.msg_namelen= sizeof(sockaddr_in);
284
285 msgs[i].msg_hdr.msg_control= ctrl.data() + i*64;
286
287 msgs[i].msg_hdr.msg_controllen = 64;
288
289 }
290
291 int r = recvmmsg(sockfd_, msgs.data(), n, 0, nullptr);
292
293 if (r < 0 && (errno == EAGAIN || errno == EWOULDBLOCK)) return 0;
294
295 if (r < 0) return -1;
296
297 return r;
298
299#else
300
301 // Fallback to single recvfrom
302
303 sockaddr_in addr{};
304
305 socklen_t alen = sizeof(addr);
306
307 ssize_t r = recvfrom(sockfd_, bufs[0].data(), bufs[0].size(), 0, (sockaddr*)&addr, &alen);
308
309 if (r < 0 && (errno == EAGAIN || errno == EWOULDBLOCK)) return 0;
310
311 if (r < 0) return -1;
312
313 return 1;
314
315#endif
316
317}

319/**
320
321* \copydoc udp::ISocket::send_batch
322
323*
324
325* @details Linux fast-path:
326
327* - Prepares `mmsghdr`/`iovec` arrays for up to `bufs.size()` messages.
328
329* - If the socket is not `connected_`, per-call `addr` is attached to each msg.
330
331* - Calls `sendmmsg()` once; on `EAGAIN`/`EWOULDBLOCK` returns 0, on other errors -1,
332
333* else the number of messages actually sent.
334
335*
336
337* Fallback:
338
339* - Loops `send()` (connected) or `sendto()` (unconnected) over `bufs`.
340
341* - Returns the **count** of successful sends.
342
343*
344
345* @note Return value is a **message count**, not bytes.
346
347*/
348
349ssize_t UdpSocket::send_batch(const std::vector<std::vector<uint8_t>>& bufs, const sockaddr_in* addr) {
350
351#if defined(__linux__)
352
353 const size_t n = bufs.size();
354
355 std::vector<iovec> iov(n);
356
357 std::vector<mmsghdr> msgs(n);
358
359 for (size_t i=0;i<n;i++) {
360
361 iov[i].iov_base = const_cast<uint8_t*>(bufs[i].data());
362
363 iov[i].iov_len = bufs[i].size();
364
365 memset(&msgs[i], 0, sizeof(mmsghdr));
366
367 msgs[i].msg_hdr.msg_iov = &iov[i];
368
369 msgs[i].msg_hdr.msg_iovlen = 1;
370
371 if (!connected_) {
372
373 msgs[i].msg_hdr.msg_name = const_cast<sockaddr_in*>(addr);
374
375 msgs[i].msg_hdr.msg_namelen = sizeof(sockaddr_in);
376
377 }
378
379 }
380
381 int r = sendmmsg(sockfd_, msgs.data(), n, 0);
382
383 if (r < 0 && (errno == EAGAIN || errno == EWOULDBLOCK)) return 0;
384
385 if (r < 0) return -1;
386
387 return r;
388
389#else
390
391 // Fallback to single sendto/connect
392
393 ssize_t cnt = 0;
394
395 for (auto& b : bufs) {
396
397 ssize_t r;
398
399 if (connected_) r = ::send(sockfd_, b.data(), b.size(), 0);
400
401 else r = ::sendto(sockfd_, b.data(), b.size(), 0, (sockaddr*)addr, sizeof(sockaddr_in));
402
403 if (r >= 0) cnt++;
404
405 }
406
407 return cnt;
408
409#endif

429/**
430
431* @brief Return up to `bufs.size()` preloaded datagrams into caller buffers.
432
433*
434
435* @details Copies from `rx_store_` into `bufs[i]` up to the minimum of source
436
437* size and destination capacity (truncation possible, consistent with UDP recv).
438
439* Advances an internal cursor so subsequent calls continue where the last ended.
440
441*
442
443* @return Number of messages copied into `bufs` (0..bufs.size()).
444
445*/
446
447ssize_t MockSocket::recv_batch(std::vector<std::vector<uint8_t>>& bufs) {
448
449 size_t i=0;
450
451 for (; i<bufs.size() && recv_cursor_ < rx_store_.size(); ++i, ++recv_cursor_) {
452
453 auto& src = rx_store_[recv_cursor_];
454
455 auto& dst = bufs[i];
456
457 size_t n = std::min(dst.size(), src.size());
458
459 std::copy(src.begin(), src.begin()+n, dst.begin());
460
461 }
462
463 return static_cast<ssize_t>(i);
464
465}
