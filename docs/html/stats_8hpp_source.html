26/**
27* @brief Key type representing a client as (IPv4 address, UDP port).
28*
29* @details
30* - @ref addr is expected to be a 32-bit IPv4 address in host byte order unless
31* otherwise noted by the caller.
32* - @ref port is the UDP port in host byte order.
33* - Equality compares both fields for exact match.
34*
35* @warning If you store network-byte-order values here, use a consistent convention
36* across the codebase and be explicit when converting (e.g., @c ntohl/ntohs).
37*/
38struct ClientKey {
39 uint32_t addr; ///< IPv4 address (host order unless documented otherwise).
40 uint16_t port; ///< UDP port (host order).
41
42 /// @brief Equality: two keys are equal iff both address and port match.
43 bool operator==(const ClientKey& o) const { return addr == o.addr && port == o.port; }
44};

46/**
47* @brief Hash functor for @ref ClientKey suitable for @c std::unordered_map.
48*
49* @details A simple, cheap mix: shift the 32-bit address and XOR with the 16-bit port.
50* This is sufficient for basic metrics/indexing. For large maps or adversarial inputs,
51* consider a stronger combiner (e.g., 64-bit splitmix or std::hash composition).
52*/
53struct ClientKeyHash {
54 /// @brief Compute a size_t hash from an address-port pair.
55 size_t operator()(const ClientKey& k) const {
56 return (static_cast<size_t>(k.addr) << 16) ^ k.port;
57 }
58};

60/**
61* @brief Aggregated counters and (optional) unique-client tracking.
62*
63* @details
64* - **Hot path:** packet/byte counters are @c std::atomic and updated with
65* @c memory_order_relaxed to avoid locks and minimize contention.
66* - **Unique clients:** tracked in an @c unordered_map guarded by a small
67* @c std::mutex. This should be called less frequently than the hot-path
68* counters to avoid contention in tight loops.
69*
70* @par Thread-safety
71* - @ref inc_sent, @ref inc_recv, @ref add_rx_bytes, @ref add_tx_bytes and the
72* getters are lock-free and thread-safe.
73* - @ref note_client and @ref unique_clients acquire an internal mutex.
74*
75* @par Consistency
76* Reads of different counters are not atomic as a group; a single @ref to_string
77* call may reflect slightly different instants per counter. This is generally fine
78* for diagnostics and metrics.
79*
80* @code
81* udp::Stats s;
82* s.inc_recv(64);
83* s.add_rx_bytes(64 * 128);
84* s.note_client(0x7f000001 /*127.0.0.1*/, 9000);
85* std::string line = s.to_string(); // "recv=64 sent=0 unique_clients=1 rx_bytes=8192 tx_bytes=0"
86* @endcode
87*/
